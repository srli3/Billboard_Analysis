#Reformat date, overwrite year value for filling the blanks
#Create a new column to store month value
#Create a new column to store year-month value

BData <- BData %>%
  mutate(date = as.Date(BData$date, format = '%m/%d/%y')) %>%
  mutate(year = year(date)) %>%
  mutate(month = month(date)) %>%
  mutate(year_month = format(BData$date, format = '%Y-%m'))

Trend of popular genre by year

genre_by_year <- BData %>%
  group_by(broad_genre, year) %>%
  summarise(year_count = n())
